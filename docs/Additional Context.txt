#!/usr/bin/env python3
"""Metric-lock the gauge-free unseeded SfM to the LiDAR (sensor_depth), the HQ counterpart
to 03_relock.py.

03_relock aligns the gauge-free BA model to an ARKit metric COLMAP target (Umeyama on camera
centers). The HQ-Depth capture has NO ARKit poses, so its ONLY absolute metric reference is the
raw LiDAR depth. This script recovers the single SfM->metric scale S directly from that depth:

  for each SfM 3D point observed in a frame, compare its SfM camera-frame depth to the LiDAR
  depth at the SAME observed pixel; S = robust median( lidar_depth / sfm_depth ).

Scale is derived per-observation (a ratio), so it is immune to the SfM gauge freedom
(rotation/translation/global-scale) — no pose is used to solve it. Then poses (tvec*=S) and
points (xyz*=S) are scaled and written as a metric COLMAP model that Stage 5 (MILo) ingests.

Writes into <session>/metric_sfm/ (does NOT touch the existing DA3 metric/); the caller swaps it in.
Run in the gs-ba env (needs pycolmap + PIL + numpy).
"""
import json
import sys
from pathlib import Path

import numpy as np
import pycolmap

REPO = Path("/home/paperspace/Documents/VS Code Projects/3D-Gaussian")
sys.path.insert(0, str(REPO))
from common import colmap_io, plyio

SESS = REPO / "sessions/session_20260704_143324"
SFM = SESS / "pose_ba" / "sfm_noseed"
OUT = SESS / "metric_sfm"
CAP = json.load(open(SESS / "capture/intrinsics.json"))
DEPTH_DIR = SESS / "capture/depth"
CONF_DIR = SESS / "capture/confidence"

color_w, color_h = CAP["color_resolution"]
depth_w, depth_h = CAP["depth_resolution"]
sx, sy = depth_w / color_w, depth_h / color_h          # color-pixel -> depth-pixel
K = np.asarray(CAP["K"], float)                          # true device K at color res
DMIN, DMAX = 0.15, 3.0

rec = pycolmap.Reconstruction(str(SFM))
print(f"[lock] SfM model: {rec.num_images()} images, {rec.num_points3D()} points")


def load_conf(fid):
    p = CONF_DIR / f"{fid}.png"
    if not p.exists():
        return None
    from PIL import Image
    a = np.asarray(Image.open(p))
    if a.ndim == 3:
        a = a[..., 0]
    return a >= 128


# ---- 1. solve the SfM->metric scale from the LiDAR ------------------------- #
ratios, per_frame = [], {}
for img in rec.images.values():
    fid = Path(img.name).stem
    dpath = DEPTH_DIR / f"{fid}.npy"
    if not dpath.exists():
        continue
    depth = np.load(dpath).astype(float)                # (240,320) meters, NaN = invalid
    conf = load_conf(fid)
    Rwc = np.asarray(img.cam_from_world.rotation.matrix(), float)
    twc = np.asarray(img.cam_from_world.translation, float)
    n = 0
    for p2d in img.points2D:
        if not p2d.has_point3D():
            continue
        X = np.asarray(rec.points3D[p2d.point3D_id].xyz, float)
        zc = float((Rwc @ X + twc)[2])
        if zc <= 1e-6:
            continue
        u, v = float(p2d.xy[0]), float(p2d.xy[1])
        ud, vd = int(round(u * sx)), int(round(v * sy))
        if not (0 <= ud < depth_w and 0 <= vd < depth_h):
            continue
        d = depth[vd, ud]
        if not np.isfinite(d) or d < DMIN or d > DMAX:
            continue
        if conf is not None and not conf[vd, ud]:
            continue
        ratios.append(d / zc)
        n += 1
    per_frame[fid] = n

ratios = np.asarray(ratios)
if ratios.size < 100:
    raise SystemExit(f"[lock] too few depth samples ({ratios.size}); cannot metric-lock")
S = float(np.median(ratios))
mad = float(np.median(np.abs(ratios - S)))
frames_hit = sum(1 for v in per_frame.values() if v > 0)
print(f"[lock] {ratios.size} (point,frame) samples over {frames_hit}/{rec.num_images()} frames")
print(f"[lock] SfM->metric scale S = {S:.6f}   MAD = {mad:.6f} ({100*mad/S:.1f}% of S)")
print(f"[lock] per-frame samples: min={min(per_frame.values())} median={int(np.median(list(per_frame.values())))} max={max(per_frame.values())}")

# ---- 2. scale + write the metric COLMAP model + init cloud ----------------- #
imgs_sorted = sorted(rec.images.values(), key=lambda im: im.name)
frame_ids, Rt_w2c, names = [], {}, {}
for im in imgs_sorted:
    fid = Path(im.name).stem
    Rwc = np.asarray(im.cam_from_world.rotation.matrix(), float)
    twc = np.asarray(im.cam_from_world.translation, float) * S
    Rt_w2c[fid] = (Rwc, twc)
    names[fid] = f"{fid}.png"
    frame_ids.append(fid)

pids = sorted(rec.points3D.keys())
points3D, xyz_all, rgb_all = {}, [], []
for i, pid in enumerate(pids, start=1):
    p = rec.points3D[pid]
    xyz = np.asarray(p.xyz, float) * S
    rgb = tuple(int(c) for c in np.asarray(p.color)[:3])
    points3D[i] = {"xyz": xyz, "rgb": rgb, "error": float(getattr(p, "error", 0.0) or 0.0), "track": []}
    xyz_all.append(xyz)
    rgb_all.append(rgb)
xyz_all = np.asarray(xyz_all)
rgb_all = np.asarray(rgb_all, dtype=np.uint8)

cameras, images = colmap_io.build_pinhole_model(
    frame_ids, {f: K for f in frame_ids}, Rt_w2c, names, (color_w, color_h), shared_intrinsics=True)

out_sparse = OUT / "colmap" / "sparse" / "0"
colmap_io.write_model(out_sparse, cameras, images, points3D)
plyio.write_ply(OUT / "points_metric.ply", xyz_all, colors=rgb_all, binary=True)

# camera-center extent sanity (feet capture: expect a handful of cm of camera motion)
centers = np.array([-Rt_w2c[f][0].T @ Rt_w2c[f][1] for f in frame_ids])
cam_extent_mm = (centers.max(0) - centers.min(0)) * 1000
obj_extent_mm = (xyz_all.max(0) - xyz_all.min(0)) * 1000
print(f"[lock] metric camera-center extent: {cam_extent_mm.round(0)} mm")
print(f"[lock] metric point-cloud extent:   {obj_extent_mm.round(0)} mm ({len(xyz_all)} pts)")

report = {
    "method": "sfm_lidar_metric_lock",
    "note": "unseeded SfM (02b) metric-locked to raw LiDAR depth (sensor_depth); HQ has no ARKit/VIO anchor",
    "sfm_source": str(SFM.relative_to(REPO)),
    "scale_S": S, "scale_MAD": mad, "scale_MAD_pct": 100 * mad / S,
    "depth_samples": int(ratios.size), "frames_used": frames_hit,
    "n_images": len(frame_ids), "n_points_init": int(len(xyz_all)),
    "camera_center_extent_mm": cam_extent_mm.tolist(),
    "point_extent_mm": obj_extent_mm.tolist(),
    "intrinsics": "device_K_shared", "convention": "world_to_camera OpenCV",
}
(OUT / "scale_report_sfm.json").write_text(json.dumps(report, indent=2))
print(f"[lock] wrote {out_sparse}")
print(f"[lock] wrote {OUT/'points_metric.ply'} + scale_report_sfm.json")

The four connective truths (why it's wired this way)
DA3's "metric" is a learned prior, not a measurement (few-% off). That is the entire reason Stage 3 exists — it re-derives true scale from physical anchors. And the known ~12% LiDAR-vs-VIO gap is the real 1 mm limiter, not pose drift. So DA3 is a geometric scaffold; scale is always re-locked before Stage 5.

Pose coherence forks the whole downstream. Coherent poses (ARKit-conditioned, or SfM) → a well-posed Stage 5. Incoherent poses (DA3 ray-pose) → Stage 5's surface machinery breaks. This fork is the HQ story — and why the SfM arm exists.

The scene must be pre-scaled to ~unit for the INRIA-lineage rasterizer (RaDe-GS/MILo overflow on ~0.1-unit metric scenes). milo_supervised does this (S=1/nerf_radius) and inverts it on output — so my metric-lock only needs metric/colmap to be truly metric, which the 1%-MAD LiDAR lock guarantees. gsplat tolerates metric scale directly — which is exactly why it's the robust default host, and a fallback worth remembering if MILo stays fragile.

Supervision is a portable recipe, not a framework. DN-Splatter gives the loss forms (EdgeAwareLogL1 depth + L1/TV normal), AGS-Mesh adds confidence-gating, StableNormal supplies the normal prior — all ported into one file (ags_depth_normal_losses.py) shared by both hosts. This is how Stage-1 LiDAR + Stage-4 normals actually enter Stage 5.

What this confirms about our live crash (the payoff)
Two independent agents (MILo internals + rasterizers) converged on the exact mechanism, and it validates everything we're doing:

MILo's learnable-SDF is periodically reset by depth-fusion: it integrates every view's rendered depth into a shared world-space pivot field, using each camera's pose. With coherent poses, a surface point back-projects to the same world spot across views → consistent TSDF votes → a clean zero-crossing → a valid mesh. With incoherent poses (DA3 ray-pose), the same point lands in different spots → contradictory front/behind votes that cancel → no zero-crossing → marching-tetrahedra emits NaN/huge verts → nvdiffrast's fixed-point tile math goes out-of-bounds → CUDA 700, sticky.

#!/usr/bin/env python3
"""Reusable reconstruction eval/compare harness (run in the pipeline_stage2_frontend env: open3d).

For each session output dir (has point_cloud.ply INRIA gaussians + optional mesh.ply):
  - stats: n_gaussians, cloud extent, object-box (subject) dims; mesh verts/faces, extent, roughness.
  - renders (Open3D offscreen, headless EGL): mesh shaded (orbit views) + cloud colored (orbit views).
  - writes a per-session montage PNG + a stats dict; a top-level montage stacks sessions for A/B.

Usage:
  eval_recon.py OUT  --label NAME  [--outdir DIR]
  eval_recon.py OUTA OUTB ... --labels A B ... --outdir DIR   # side-by-side comparison
"""
import argparse
import json
from pathlib import Path

import numpy as np
import open3d as o3d

SH_C0 = 0.28209479177387814


# ---------- IO ----------
def read_inria_ply(path):
    """Return xyz(N,3), rgb(N,3 in [0,1] from SH DC), opacity(N)."""
    with open(path, "rb") as f:
        assert f.readline().strip() == b"ply"
        f.readline()  # format
        names, n = [], 0
        while True:
            ln = f.readline().decode("latin1").strip()
            if ln.startswith("element vertex"):
                n = int(ln.split()[-1])
            elif ln.startswith("property"):
                names.append(ln.split()[-1])
            elif ln == "end_header":
                break
        data = np.frombuffer(f.read(n * len(names) * 4), np.float32).reshape(n, len(names))
    idx = {nm: i for i, nm in enumerate(names)}
    xyz = data[:, [idx["x"], idx["y"], idx["z"]]].astype(np.float64)
    if "f_dc_0" in idx:
        fdc = data[:, [idx["f_dc_0"], idx["f_dc_1"], idx["f_dc_2"]]]
        rgb = np.clip(SH_C0 * fdc + 0.5, 0, 1).astype(np.float64)
    else:
        rgb = np.full((n, 3), 0.6)
    op = 1.0 / (1.0 + np.exp(-data[:, idx["opacity"]])) if "opacity" in idx else np.ones(n)
    return xyz, rgb, op


def object_box(xyz, pct=1.5, pad=0.05):
    lo = np.percentile(xyz, pct, 0)
    hi = np.percentile(xyz, 100 - pct, 0)
    p = pad * (hi - lo)
    return lo - p, hi + p


# ---------- rendering ----------
def _orbit_eyes(center, radius, n=3, elev=0.55):
    eyes = []
    for az in np.linspace(0, 2 * np.pi, n, endpoint=False):
        d = np.array([np.cos(az), np.sin(az), elev], float)
        eyes.append(center + d / np.linalg.norm(d) * radius * 2.3)
    return eyes


def render_geometry(geom, shader, point_size=3.0, W=760, H=620, n_views=3, bg=(1, 1, 1)):
    """Render a geometry from n_views orbit angles; return a horizontally-stacked uint8 image."""
    aabb = geom.get_axis_aligned_bounding_box()
    center = aabb.get_center()
    radius = float(np.linalg.norm(aabb.get_extent())) * 0.5 + 1e-6
    up = [0, 0, 1]
    r = o3d.visualization.rendering.OffscreenRenderer(W, H)
    r.scene.set_background([*bg, 1.0])
    mat = o3d.visualization.rendering.MaterialRecord()
    mat.shader = shader
    if shader == "defaultUnlit":
        mat.point_size = point_size
    r.scene.add_geometry("g", geom, mat)
    r.scene.scene.set_sun_light([0.3, 0.3, -0.8], [1, 1, 1], 90000)
    r.scene.scene.enable_sun_light(True)
    tiles = []
    for eye in _orbit_eyes(center, radius, n_views):
        r.setup_camera(55.0, center.tolist(), eye.tolist(), up)
        tiles.append(np.asarray(r.render_to_image()))
    return np.concatenate(tiles, axis=1)


def mesh_roughness_deg(mesh):
    """Mean dihedral angle between adjacent triangles (deg) — higher = bumpier. Open3D-only."""
    mesh.compute_triangle_normals()
    tn = np.asarray(mesh.triangle_normals)
    tris = np.asarray(mesh.triangles)
    # edge -> adjacent triangle list
    from collections import defaultdict
    edge_tris = defaultdict(list)
    for ti, t in enumerate(tris):
        for a, b in ((t[0], t[1]), (t[1], t[2]), (t[2], t[0])):
            edge_tris[(min(a, b), max(a, b))].append(ti)
    ang = []
    for ts in edge_tris.values():
        if len(ts) == 2:
            d = np.clip(np.dot(tn[ts[0]], tn[ts[1]]), -1, 1)
            ang.append(np.degrees(np.arccos(d)))
    return float(np.mean(ang)) if ang else float("nan")


# ---------- per-session ----------
def eval_session(out_dir, label, render_dir):
    out_dir = Path(out_dir)
    stats = {"label": label, "dir": str(out_dir)}
    strips = []

    cloud_ply = out_dir / "point_cloud.ply"
    if cloud_ply.exists():
        xyz, rgb, op = read_inria_ply(cloud_ply)
        ext = (xyz.max(0) - xyz.min(0)) * 1000
        lo, hi = object_box(xyz)
        obj_dims = (hi - lo) * 1000
        stats["cloud"] = {"n_gaussians": int(len(xyz)),
                          "extent_mm": [round(float(v), 1) for v in ext],
                          "object_box_mm": [round(float(v), 1) for v in obj_dims]}
        # crop to subject for a cleaner render
        m = np.all((xyz >= lo) & (xyz <= hi), 1)
        pc = o3d.geometry.PointCloud()
        pc.points = o3d.utility.Vector3dVector(xyz[m])
        pc.colors = o3d.utility.Vector3dVector(rgb[m])
        strips.append(("cloud", render_geometry(pc, "defaultUnlit", point_size=3.5)))

    mesh_ply = out_dir / "mesh.ply"
    if mesh_ply.exists():
        mesh = o3d.io.read_triangle_mesh(str(mesh_ply))
        mesh.compute_vertex_normals()
        v = np.asarray(mesh.vertices)
        ext = (v.max(0) - v.min(0)) * 1000
        stats["mesh"] = {"vertices": int(len(v)), "faces": int(len(mesh.triangles)),
                         "extent_mm": [round(float(x), 1) for x in ext],
                         "roughness_dihedral_deg": round(mesh_roughness_deg(mesh), 2)}
        strips.append(("mesh", render_geometry(mesh, "defaultLit")))

    # stack this session's strips vertically, save
    if strips:
        H = max(s.shape[0] for _, s in strips)
        W = max(s.shape[1] for _, s in strips)
        rows = []
        for _, s in strips:
            pad = np.full((s.shape[0], W - s.shape[1], 3), 255, np.uint8) if s.shape[1] < W else None
            rows.append(np.concatenate([s, pad], 1) if pad is not None else s)
        montage = np.concatenate(rows, 0)
        p = Path(render_dir) / f"recon_{label}.png"
        o3d.io.write_image(str(p), o3d.geometry.Image(montage.astype(np.uint8)))
        stats["montage"] = str(p)
    return stats, montage if strips else None


def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("outdirs", nargs="+")
    ap.add_argument("--labels", nargs="+")
    ap.add_argument("--outdir", default=".")
    args = ap.parse_args()
    labels = args.labels or [Path(o).parent.name for o in args.outdirs]
    Path(args.outdir).mkdir(parents=True, exist_ok=True)

    all_stats, montages = [], []
    for od, lab in zip(args.outdirs, labels):
        s, mont = eval_session(od, lab, args.outdir)
        all_stats.append(s)
        if mont is not None:
            montages.append((lab, mont))
        print(f"[eval] {lab}: {json.dumps({k: v for k, v in s.items() if k in ('cloud', 'mesh')})}")

    if len(montages) > 1:
        W = max(m.shape[1] for _, m in montages)
        rows = []
        for lab, m in montages:
            if m.shape[1] < W:
                m = np.concatenate([m, np.full((m.shape[0], W - m.shape[1], 3), 255, np.uint8)], 1)
            rows.append(m)
        comp = np.concatenate(rows, 0)
        cp = Path(args.outdir) / "comparison.png"
        o3d.io.write_image(str(cp), o3d.geometry.Image(comp.astype(np.uint8)))
        print(f"[eval] comparison montage -> {cp}")
    Path(args.outdir, "stats.json").write_text(json.dumps(all_stats, indent=2))
    print(f"[eval] stats -> {Path(args.outdir, 'stats.json')}")


if __name__ == "__main__":
    main()
